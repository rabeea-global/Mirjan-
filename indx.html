import ezdxf from ezdxf import colors def setup_layers(doc): """Establish KBR-standard layering system""" doc.layers.add("S-ANNO-TTLB", color=colors.BLUE) doc.layers.add("S-CONC-OUTL", color=colors.CYAN) doc.layers.add("S-REBR-MAIN", color=colors.RED) doc.layers.add("S-REBR-TIES", color=colors.YELLOW) doc.layers.add("S-ANNO-TEXT", color=colors.WHITE) def draw_title_block(msp): """Generates a highly professional title block""" # Border msp.add_lwpolyline([(0, 0), (8.41, 0), (8.41, 5.94), (0, 5.94)], close=True, dxfattribs={"layer": "S-ANNO-TTLB"}) # Title Box msp.add_lwpolyline([(6.0, 0), (6.0, 1.5), (8.41, 1.5)], dxfattribs={"layer": "S-ANNO-TTLB"}) # Text Entries msp.add_text("MVC DESIGN OFFICE", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.15}).set_placement((6.2, 1.2)) msp.add_text("PROJECT: COMMERCIAL RESTAURANT GRID", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.08}).set_placement((6.2, 0.9)) msp.add_text("DWG TITLE: STRUCTURAL SHOP DRAWINGS", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.08}).set_placement((6.2, 0.7)) msp.add_text("SCALE: NTS | UNITS: METERS / KG", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.06}).set_placement((6.2, 0.5)) def draw_column_section(msp, x_offset, y_offset): """Draws the 0.3 x 0.5m Edge Column with 10 bars of 22mm""" b, h, cover = 0.3, 0.5, 0.04 # Concrete Outline msp.add_lwpolyline([ (x_offset, y_offset), (x_offset + b, y_offset), (x_offset + b, y_offset + h), (x_offset, y_offset + h) ], close=True, dxfattribs={"layer": "S-CONC-OUTL"}) # Tie / Stirrup (Accounting for 40mm cover) tie_b, tie_h = b - 2*cover, h - 2*cover tie_x, tie_y = x_offset + cover, y_offset + cover msp.add_lwpolyline([ (tie_x, tie_y), (tie_x + tie_b, tie_y), (tie_x + tie_b, tie_y + tie_h), (tie_x, tie_y + tie_h) ], close=True, dxfattribs={"layer": "S-REBR-TIES"}) # Rebar Details (10 bars) - 5 on each short face for moment resistance bar_radius = 0.011 # 22mm diameter spacing_y = tie_h / 4 for i in range(5): msp.add_circle((tie_x + bar_radius, tie_y + (i * spacing_y)), bar_radius, dxfattribs={"layer": "S-REBR-MAIN"}) msp.add_circle((tie_x + tie_b - bar_radius, tie_y + (i * spacing_y)), bar_radius, dxfattribs={"layer": "S-REBR-MAIN"}) # Labels msp.add_text("SECTION A-A: EDGE COLUMN (0.3 x 0.5 m)", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset, y_offset - 0.15)) msp.add_text("MAIN: 10 T22 | TIES: T10 @ 100 mm", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.04}).set_placement((x_offset, y_offset - 0.22)) def draw_beam_section(msp, x_offset, y_offset): """Draws the 0.3 x 0.5m Cantilever Beam Section""" b, h, cover = 0.3, 0.5, 0.04 # Concrete Outline msp.add_lwpolyline([ (x_offset, y_offset), (x_offset + b, y_offset), (x_offset + b, y_offset + h), (x_offset, y_offset + h) ], close=True, dxfattribs={"layer": "S-CONC-OUTL"}) # Tie / Stirrup tie_b, tie_h = b - 2*cover, h - 2*cover tie_x, tie_y = x_offset + cover, y_offset + cover msp.add_lwpolyline([ (tie_x, tie_y), (tie_x + tie_b, tie_y), (tie_x + tie_b, tie_y + tie_h), (tie_x, tie_y + tie_h) ], close=True, dxfattribs={"layer": "S-REBR-TIES"}) # Main Top Steel: 5 T25 (In tension zone - arranged in one row for simplification) bar_rad_25 = 0.0125 spacing_x = tie_b / 4 for i in range(5): msp.add_circle((tie_x + (i * spacing_x), tie_y + tie_h - bar_rad_25), bar_rad_25, dxfattribs={"layer": "S-REBR-MAIN"}) # Main Bottom Steel: 2 T16 (Compression/hanger bars) bar_rad_16 = 0.008 msp.add_circle((tie_x + bar_rad_16, tie_y + bar_rad_16), bar_rad_16, dxfattribs={"layer": "S-REBR-MAIN"}) msp.add_circle((tie_x + tie_b - bar_rad_16, tie_y + bar_rad_16), bar_rad_16, dxfattribs={"layer": "S-REBR-MAIN"}) # Labels msp.add_text("SECTION B-B: CANTILEVER BEAM (0.3 x 0.5 m)", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset, y_offset - 0.15)) msp.add_text("TOP: 5 T25 | BOT: 2 T16 | TIES: T10 @ 100 mm", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.04}).set_placement((x_offset, y_offset - 0.22)) def draw_bbs_table(msp, x_offset, y_offset): """Draws the Bar Bending Schedule (BBS) Table in kg""" # Table Outline msp.add_lwpolyline([(x_offset, y_offset), (x_offset+4, y_offset), (x_offset+4, y_offset+1.5), (x_offset, y_offset+1.5)], close=True, dxfattribs={"layer": "S-ANNO-TEXT"}) # Headers msp.add_text("BAR BENDING SCHEDULE (BBS)", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.06}).set_placement((x_offset + 0.1, y_offset + 1.35)) msp.add_text("MEMBER | MAIN (kg) | TIES (kg) | TOTAL (kg)", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset + 0.1, y_offset + 1.15)) msp.add_line((x_offset, y_offset+1.1), (x_offset+4, y_offset+1.1), dxfattribs={"layer": "S-ANNO-TEXT"}) # Data Rows msp.add_text("SLAB (200mm) | 4190 | --- | 4190", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset + 0.1, y_offset + 0.9)) msp.add_text("BEAMS (INT) | 935 | 530 | 1465", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset + 0.1, y_offset + 0.7)) msp.add_text("CANTILEVERS | 1225 | 225 | 1450", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset + 0.1, y_offset + 0.5)) msp.add_text("COLS (EDGE) | 955 | 180 | 1135", dxfattribs={"layer": "S-ANNO-TEXT", "height": 0.05}).set_placement((x_offset + 0.1, y_offset + 0.3)) def generate_shop_drawing(): """Main execution block""" doc = ezdxf.new("R2010") setup_layers(doc) msp = doc.modelspace() draw_title_block(msp) draw_column_section(msp, x_offset=1.0, y_offset=4.0) draw_beam_section(msp, x_offset=3.0, y_offset=4.0) draw_bbs_table(msp, x_offset=1.0, y_offset=1.0) doc.saveas("MVC_Structural_Shop_Drawing.dxf") print("SUCCESS: Shop Drawing Generated as 'MVC_Structural_Shop_Drawing.dxf'") if __name__ == "__main__": generate_shop_drawing()
